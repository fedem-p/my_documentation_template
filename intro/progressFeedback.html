def oneStepSearch(inp):
    x_range = inp[0]
    y_range = inp[1]
    z_range = inp[2]
    x_bool = True

    arra = []

    for z in range(z_range):
        if z%2==0:
            for y in range(0, y_range+1, 1):
                if x_bool:
                    for x in range(0, x_range+1, 1):
    #                         print("z = {}, y = {}, x = {}".format(z,y,x))
                            arra.append([z,y,x])
                    x_bool = False
                else:
                    for x in range(x_range, -1, -1):
    #                         print("z = {}, y = {}, x = {}".format(z,y,x))
                            arra.append([z,y,x])
                    x_bool = True

        else:
            for y in range(y_range, -1, -1):
                if x_bool:
                    for x in range(0, x_range+1, 1):
    #                         print("z = {}, y = {}, x = {}".format(z,y,x))
                            arra.append([z,y,x])
                    x_bool = False
                else:
                    for x in range(x_range, -1, -1):
    #                         print("z = {}, y = {}, x = {}".format(z,y,x))
                            arra.append([z,y,x])
                    x_bool = True
    return np.array(arra)

def isOneStep(arr):
    L = len(arr)
    m = 0 # init maximum
    
    for i in range(L-1):
        k = np.sum(np.absolute(np.subtract(arr[i], arr[i+1])))
        if k>m:
            m=k
    if m>1:
        return False
    else:
        return True
